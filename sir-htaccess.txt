# BEGIN GD-SSL
<IfModule mod_rewrite.c>

RewriteEngine On
RewriteCond %{HTTPS} !=on
RewriteCond %{HTTP_USER_AGENT} ^(.+)$
RewriteCond %{SERVER_NAME} ^aspcares\.com$
RewriteRule .* https://%{SERVER_NAME}%{REQUEST_URI} [R=301,L]

</IfModule>
